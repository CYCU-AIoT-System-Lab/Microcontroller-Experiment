50__STATIC_FORCEINLINE void __get_PAC_KEY_P (uint32_t* pPacKey) {
51 __ASM volatile (
52 "mrs r1, pac_key_p_0\n"
53 "str r1,[%0,#0]\n"
54 "mrs r1, pac_key_p_1\n"
55 "str r1,[%0,#4]\n"
