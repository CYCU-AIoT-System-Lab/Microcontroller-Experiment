141u32 SPI_FLASH_ReadJEDECID_0(void)
142{
143 u32 Temp = 0, Temp0 = 0, Temp1 = 0, Temp2 = 0;
144
145 /* Select the SPI FLASH */
146 SPI_FLASH_SEL_ACTIVE();
147
148 /* Send "JEDEC Read-ID" instruction */
149 SPI_FLASH_SendByte_0(SPI_FLASH_CMD_JEDECID_0);
150
151 /* Read a byte from the FLASH */
152 Temp0 = SPI_FLASH_SendByte_0(SPI_FLASH_DUMMY_BYTE);
153
154 /* Read a byte from the FLASH */
155 Temp1 = SPI_FLASH_SendByte_0(SPI_FLASH_DUMMY_BYTE);
156

158 Temp2 = SPI_FLASH_SendByte_0(SPI_FLASH_DUMMY_BYTE);
159
160 /* Deselect the SPI FLASH */
161 SPI_FLASH_SEL_INACTIVE();
162
163 Temp = (Temp0 << 16) | (Temp1 << 8) | Temp2;
