222void ADC_RegularChannelConfig(HT_ADC_TypeDef* HT_ADCn, u8 ADC_CH_n, u8 Rank, u8 SampleClock)
223{
224 u32 tmpreg1, tmpreg2;
225
226 /* Check the parameters */
227 Assert_Param(IS_ADC(HT_ADCn));
228 Assert_Param(IS_ADC_CHANNEL(ADC_CH_n));
229 Assert_Param(IS_ADC_REGULAR_RANK(Rank));
230 Assert_Param(IS_ADC_INPUT_SAMPLING_TIME(SampleClock));
231
232 /* config sampling clock of correspond ADC input channel */
233 HT_ADCn->STR[ADC_CH_n] = SampleClock;
234
235 /* Get the old register value */
236 tmpreg1 = HT_ADCn->LST[Rank >> 2];
237 /* Calculate the mask to clear */
238 tmpreg2 = LST_SEQ_SET << (8 * (Rank & 0x3));
239 /* Clear the old SEQx bits for the selected rank */
240 tmpreg1 &= ~tmpreg2;
241 /* Calculate the mask to set */
242 tmpreg2 = (u32)ADC_CH_n << (8 * (Rank & 0x3));
243 /* Set the SEQx bits for the selected rank */
244 tmpreg1 |= tmpreg2;
245 /* Store the new register value */
246 HT_ADCn->LST[Rank >> 2] = tmpreg1;
247}
