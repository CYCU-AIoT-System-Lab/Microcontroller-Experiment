54void MX25L1606E_PageWrite(u8* WriteBuffer, u32 WriteAddr, u32 NumOfWriteByte)
55{
56 /* Enable the write access to the FLASH */
57 SPI_FLASH_WriteEnable_0();
58
59 /* Select the SPI FLASH */
60 SPI_FLASH_SEL_ACTIVE();
61 /* Send "Page Program" instruction */
62 SPI_FLASH_SendByte_0(MX25L1606E_CMD_WRITE);
63 /* Send high byte address of WriteAddr */
64 SPI_FLASH_SendByte_0((WriteAddr & 0xFF0000) >> 16);
65 /* Send medium byte address of WriteAddr */
66 SPI_FLASH_SendByte_0((WriteAddr & 0xFF00) >> 8);
67 /* Send low byte address of WriteAddr */
68 SPI_FLASH_SendByte_0(WriteAddr & 0xFF);
69
70 while (NumOfWriteByte--)
71 {
72 /* Send the current byte */
73 SPI_FLASH_SendByte_0(*WriteBuffer);
74 /* Point on the next byte to be written */
75 WriteBuffer++;
76 }
77
78 /* Deselect the SPI FLASH */
79 SPI_FLASH_SEL_INACTIVE();
80
81 /* Wait the end of Flash writing */
82 SPI_FLASH_WaitForWriteEnd_0();
83}
