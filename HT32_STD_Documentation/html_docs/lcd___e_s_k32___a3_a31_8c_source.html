376void LCD_show_Matrix(u8 StartX, u8 StartY, LCD_PIXEL_CONTROL_Enum on_off)
377{
378 u8 bit_index, ram_index;
379
380 if (StartX >= _MATRIX_X_SIZE)
381 StartX = _MATRIX_X_SIZE-1;
382
383 if (StartY >= _MATRIX_Y_SIZE)
384 StartY = _MATRIX_Y_SIZE-1;
385
386 bit_index = TAB_MATRIX_X_BIT_INDEX[StartX];
387 ram_index = TAB_MATRIX_Y_RAM_INDEX[StartY] + (bit_index >> 5);
388 bit_index = bit_index & 0x1F;
389
390 if (on_off)
391 HT_LCD->RAM[ram_index] |= (u32)0x1 << bit_index;
392 else
393 HT_LCD->RAM[ram_index] &= ~((u32)0x1 << bit_index);
394}

433void LCD_show_Picture(u8 StartX, u8 StartY, uc8 pPicture[])
434{
435 u8 W = pPicture[0];
