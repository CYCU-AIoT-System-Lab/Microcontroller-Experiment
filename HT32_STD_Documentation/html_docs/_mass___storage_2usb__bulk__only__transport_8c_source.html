213void BOT_OUTProcess(void)
214{
215 BOT_Media_TypeDef *pMedia = GET_MEDIA(gCBW.bCBWLUN);
216
217 switch (gBOTState)
218 {
219 /*------------------------------------------------------------------------------------------------------*/
220 /* Decode and process CBW */
221 /*------------------------------------------------------------------------------------------------------*/
222 case BOT_STATE_IDLE:
223 {
224 _BOT_CBWProcess();
225 break;
226 }
227 /*------------------------------------------------------------------------------------------------------*/
228 /* Data OUT process */
229 /*------------------------------------------------------------------------------------------------------*/
230 case BOT_STATE_DATAOUT:
231 {
232
233 SBC_DataOUT(pMedia, gCBW.CBWCB);
234 break;
235 }
236 /*------------------------------------------------------------------------------------------------------*/
237 /* BOT State machine out of sync, report Phase Error by CSW */
238 /*------------------------------------------------------------------------------------------------------*/
239 default:
240 {
241 SBC_UpdateSenseData(pMedia, SK_ILLEGAL_REQUEST, ASC_INVALID_FIELED_IN_COMMAND);
242 BOT_ErrorHandler(BOT_STALL_BOTH, BOT_CSW_STATUS_PHASEERR);
243 break;
244 }
245 }
246
247 return;
248}

254void BOT_INProcess(void)
255{
256 BOT_Media_TypeDef *pMedia = GET_MEDIA(gCBW.bCBWLUN);
257
258 switch (gBOTState)
259 {
260 /*------------------------------------------------------------------------------------------------------*/
261 /* Data IN process */
262 /*------------------------------------------------------------------------------------------------------*/
263 case BOT_STATE_DATAIN:
264 {
265 SBC_DataIN(pMedia, gCBW.CBWCB);
266 break;
267 }
268 /*------------------------------------------------------------------------------------------------------*/
269 /* Last Data IN, send CSW */
270 /*------------------------------------------------------------------------------------------------------*/
271 case BOT_STATE_DATAIN_LAST:
272 {
273 /* STALL IN Endpoint when last Data IN but Data Residue in CSW is not zero */
274 if (gCSW.dCSWDataResidue != 0)
275 {
276 _BOT_StallEPT(BOT_STALL_IN);
277 }
278 BOT_SendCSW(BOT_CSW_STATUS_PASSED);
279 break;
280 }
281 /*------------------------------------------------------------------------------------------------------*/
282 /* CSW is sent */
283 /*------------------------------------------------------------------------------------------------------*/
284 case BOT_STATE_SENDCSW:
285 //case BOT_STATE_ERROR:
286 {
287 gBOTState = BOT_STATE_IDLE;
288 /*----------------------------------------------------------------------------------------------------*/
289 /* Read OUT data with zero length to inform USB Device peripheral that process of CBW OUT */
290 /* data is finished. The USB Device will send ACK to Host when received next CBW OUT data */
291 /*----------------------------------------------------------------------------------------------------*/
292 USBDCore_EPTReadOUTData(BOT_OUT_EPT, (u32 *)NULL, USB_DISCARD_OUT_DATA);
293 break;
294 }
295 default:
296 {
297 break;
298 }
299 }
300
301 return;
302}

313u32 BOT_CheckCBWParameter(BOT_Dir_Enum DeviceDirection, u32 uDeviceLength)
314{
315 BOT_CSW_STATUS_Enum status = BOT_CSW_STATUS_PASSED;
316 BOT_STALL_Enum direction;
317 u32 Result;
318
319 __DBG_USBPrintf("%06ld %dH%d %dD%d\r\n", __DBG_USBCount,
320 gCBW.bmCBWFlags.Direction,
321 gCBW.dCBWDataTransferLength,
322 DeviceDirection,
323 uDeviceLength
324 );
325
326 /*--------------------------------------------------------------------------------------------------------*/
327 /* Case 1 ~ 3 */
328 /*--------------------------------------------------------------------------------------------------------*/
329 if (gCBW.dCBWDataTransferLength == 0)
330 {
331 if (DeviceDirection == BOT_DIR_NONE)
332 {
333 /*----------------------------------------------------------------------------------------------------*/
334 /* Case 1 (Hn = Dn) */
335 /*----------------------------------------------------------------------------------------------------*/
336 __DBG_USBPrintf("%06ld CASE1\r\n", __DBG_USBCount);
337 gCSW.dCSWDataResidue = 0;
338 }
339 else
340 {
341 /*----------------------------------------------------------------------------------------------------*/
342 /* Case 2 (Hn < Di) / Case 3 (Hn < Do) */
343 /*----------------------------------------------------------------------------------------------------*/
344 direction = BOT_STALL_NONE;
345 status = BOT_CSW_STATUS_PHASEERR;
346 __DBG_USBPrintf("%06ld CASE2, 3\r\n", __DBG_USBCount);
347 }
348 }
349 /*--------------------------------------------------------------------------------------------------------*/
350 /* Case 6 (Hi = Di) / Case 12 (Ho = Do) */
351 /*--------------------------------------------------------------------------------------------------------*/
352 else if (gCBW.dCBWDataTransferLength == uDeviceLength && gCBW.bmCBWFlags.Direction == DeviceDirection)
353 {
354 gBOTState = (DeviceDirection == BOT_DIR_IN) ? BOT_STATE_DATAIN : BOT_STATE_DATAOUT;
355 __DBG_USBPrintf("%06ld CASE6, 12\r\n", __DBG_USBCount);
356 }
357 /*--------------------------------------------------------------------------------------------------------*/
358 /* Case 4, 5, 7, 8 */
359 /*--------------------------------------------------------------------------------------------------------*/
360 else if (gCBW.bmCBWFlags.Direction == BOT_DIR_IN)
361 {
362 switch (DeviceDirection)
363 {
364 case BOT_DIR_NONE:
365 case BOT_DIR_IN:
366 {
367 if (gCBW.dCBWDataTransferLength > uDeviceLength)
368 {
369 /*------------------------------------------------------------------------------------------------*/
370 /* Case 4 (Hi > Dn) / Case 5 (Hi > Di) */
371 /*------------------------------------------------------------------------------------------------*/
372 direction = BOT_STALL_IN;
373 status = BOT_CSW_STATUS_FAILED;
374 gCSW.dCSWDataResidue = gCBW.dCBWDataTransferLength - uDeviceLength;
375 __DBG_USBPrintf("%06ld CASE4, 5\r\n", __DBG_USBCount);
376 }
377 else if (gCBW.dCBWDataTransferLength < uDeviceLength)
378 {
379 /*------------------------------------------------------------------------------------------------*/
380 /* Case 7 (Hi < Di) */
381 /*------------------------------------------------------------------------------------------------*/
382 direction = BOT_STALL_IN;
383 status = BOT_CSW_STATUS_PHASEERR;
384 __DBG_USBPrintf("%06ld CASE7\r\n", __DBG_USBCount);
385 }
386 break;
387 }
388 case BOT_DIR_OUT:
389 {
390 /*--------------------------------------------------------------------------------------------------*/
391 /* Case 8 (Hi <> Do) */
392 /*--------------------------------------------------------------------------------------------------*/
393 direction = BOT_STALL_IN;
394 status = BOT_CSW_STATUS_PHASEERR;
395 __DBG_USBPrintf("%06ld CASE8\r\n", __DBG_USBCount);
396 break;
397 }
398 }
399 }
400 /*--------------------------------------------------------------------------------------------------------*/
401 /* Case 9, 10, 11, 13 */
402 /*--------------------------------------------------------------------------------------------------------*/
403 else if (gCBW.bmCBWFlags.Direction == BOT_DIR_OUT)
404 {
405 switch (DeviceDirection)
406 {
407 case BOT_DIR_NONE:
408 case BOT_DIR_OUT:
409 {
410 if (gCBW.dCBWDataTransferLength > uDeviceLength)
411 {
412 /*------------------------------------------------------------------------------------------------*/
413 /* Case 9 (Ho > Dn) / Case 11 (Ho > Do) */
414 /*------------------------------------------------------------------------------------------------*/
415 direction = BOT_STALL_OUT;
416 status = BOT_CSW_STATUS_FAILED;
417 gCSW.dCSWDataResidue = gCBW.dCBWDataTransferLength - uDeviceLength;
418 __DBG_USBPrintf("%06ld CASE9, 11\r\n", __DBG_USBCount);
419 }
420 else if (gCBW.dCBWDataTransferLength < uDeviceLength)
421 {
422 /*------------------------------------------------------------------------------------------------*/
423 /* Case 13 (Ho < Do) */
424 /*------------------------------------------------------------------------------------------------*/
425 direction = BOT_STALL_OUT;
426 status = BOT_CSW_STATUS_PHASEERR;
427 __DBG_USBPrintf("%06ld CASE13\r\n", __DBG_USBCount);
428 }
429 break;
430 }
431 case BOT_DIR_IN:
432 {
433 /*--------------------------------------------------------------------------------------------------*/
434 /* Case 10 (Ho <> Di) */
435 /*--------------------------------------------------------------------------------------------------*/
436 direction = BOT_STALL_OUT;
437 status = BOT_CSW_STATUS_PHASEERR;
438 __DBG_USBPrintf("%06ld CASE10\r\n", __DBG_USBCount);
439 break;
440 }
441 }
442 }
443
444 if (status != BOT_CSW_STATUS_PASSED)
445 {
446 SBC_UpdateSenseData(GET_MEDIA(gCBW.bCBWLUN), SK_ILLEGAL_REQUEST, ASC_INVALID_FIELED_IN_COMMAND);
447 BOT_ErrorHandler(direction, status);
448 Result = FALSE;
449 }
450 else
451 {
452 Result = TRUE;
453 }
454
455 return Result;
456}

560static void _BOT_CBWProcess(void)
561{
562 extern USBDCore_TypeDef gUSBCore;
563 BOT_Media_TypeDef *pMedia;
564
565 u32 uCBWLen;
566
567 /*--------------------------------------------------------------------------------------------------------*/
568 /* Read CBW OUT data from USB Buffer. */
569 /*--------------------------------------------------------------------------------------------------------*/
570 /* Cause the address-of-packed-member warning of GNU_ARM, */
571 /* skip by #pragma GCC diagnostic ignored "-Waddress-of-packed-member". */
572 uCBWLen = USBDCore_EPTReadMemory(BOT_OUT_EPT, (u32 *)&gCBW, BOT_MAX_PACKET_SIZE);
573
574 __DBG_USBPrintf("%06ld CBW \t[%x][%d][%d][%d][%d]\r\n", ++__DBG_USBCount, gCBW.dCBWTag, gCBW.dCBWDataTransferLength, gCBW.bmCBWFlags.Direction, gCBW.bCBWLUN, gCBW.bCBWCBLength);
575 __DBG_USBDump((uc8 *)&gCBW, uCBWLen);
576
577 gCSW.dCSWTag = gCBW.dCBWTag;
578
579 pMedia = (gCBW.bCBWLUN < BOT_LUN_NUM) ? GET_MEDIA(gCBW.bCBWLUN) : NULL;
580
581 /*--------------------------------------------------------------------------------------------------------*/
582 /* Check CBW is valid or not */
583 /*--------------------------------------------------------------------------------------------------------*/
584 if (gCBW.dCBWSignature != _BOT_CBW_SIGNATURE || uCBWLen != _BOT_CBW_PACKET_LENGTH)
585 {
586 _BOT_StallEPT(BOT_STALL_BOTH);
587 IS_WAIT_RESET_RECOVERY = TRUE;
588 gBOTState = BOT_STATE_ERROR;
589 return;
590 }
591
592 gCSW.dCSWDataResidue = gCBW.dCBWDataTransferLength;
593
594 /*--------------------------------------------------------------------------------------------------------*/
595 /* Check CBW is meaningful or not */
596 /*--------------------------------------------------------------------------------------------------------*/
597 if ((gCBW.bCBWLUN >= BOT_LUN_NUM) || (gCBW.bCBWCBLength < 1) || (gCBW.bCBWCBLength > _BOT_MAX_CB_LENGTH))
598 {
599 _BOT_StallEPT(BOT_STALL_BOTH);
600 IS_WAIT_RESET_RECOVERY = TRUE;
601 gBOTState = BOT_STATE_ERROR;
602 return;
603 }
604
605 API_USB_EPTn_CLR_HALT(BOT_OUT_EPT);
606 API_USB_EPTn_CLR_HALT(BOT_IN_EPT);
607 SBC_CMDHandler(pMedia, (u8 *)gCBW.CBWCB);
608
609 return;
610}
